\section{Problem \& Motivation}

\begin{frame}{The problem}
  \textbf{One hospital, one week. An elective waiting list always bigger
  than the week's OR capacity.}
  \begin{itemize}\small
    \item For each waiting case: which room, which day, what clock time,
      which surgeon -- or defer to the following week.
    \item \textbf{Feasibility} constraint: rooms, surgeons, shared
      equipment, and downstream recovery beds are all finite and shared.
    \item \textbf{Fairness} constraint: clinical urgency, not FIFO order,
      should determine priority -- a genuinely urgent patient must not sit
      behind a routine one merely because of arrival order.
    \item \textbf{Scope (Cardoen et al., 2010):} this is \emph{advance
      scheduling} -- assigning cases to days/rooms ahead of the session --
      distinct from same-day sequencing or real-time disruption handling.
  \end{itemize}
\end{frame}

\begin{frame}{Why it matters}
  \begin{itemize}\small
    \item \textbf{OR time is expensive and perishable.}
      $\sim\$30$--$40$/minute once staffing, equipment and overhead
      are amortised (Macario, 2010). An idle OR-minute today cannot be
      banked for tomorrow.
    \item \textbf{The waiting list is the audited metric.}
      In systems like Portugal's SIGIC, the UK NHS RTT, and Canadian
      provincial benchmarks, waiting-list breach rates are the number
      that gets published and that planners are graded on -- not just a
      side metric.
    \item \textbf{Elective and emergency demand share the same rooms}
      (Van Riet \& Demeulemeester, 2015), making capacity allocation a
      genuine strategic decision, not a by-product of scheduling.
    \item \textbf{Evidence from the literature:} Cardoen, Demeulemeester
      \& Beli\"en (2010) survey over 200 papers; advance scheduling
      under priority/deadline constraints is its own well-defined
      sub-problem, with this project occupying its core scope.
  \end{itemize}
\end{frame}

\begin{frame}{The Portuguese case: the SIGIC system}
  \textbf{SIGIC} -- \emph{Sistema Integrado de Gest\~ao de Inscritos
  para Cirurgia} -- is Portugal's national waiting-list management
  system, established by \textbf{Portaria n.\textsuperscript{o} 45/2008}
  (Di\'ario da Rep\'ublica).

  \vspace{0.5em}
  Every elective surgical case is assigned one of \textbf{four clinical
  priority tiers}, each with its own \textbf{maximum acceptable wait}:

  \vspace{0.4em}
  \begin{center}\small
  \begin{tabular}{clc}
    \toprule
    Priority & Clinical meaning & Max wait (days) \\
    \midrule
    1 & Routine elective & 270 \\
    2 & Elevated urgency & 60 \\
    3 & Urgent & 15 \\
    4 & Emergent add-on -- run this week & 3 \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.4em}
  \footnotesize These four numbers are \emph{not} illustrative defaults
  invented for this demo -- they are the model's $\text{wl}^{max}_p$
  parameters, taken directly from a real, legislated, audited policy.
\end{frame}

\begin{frame}{The Portuguese case: what the 2016 CHLN audit found}
  Marques \& Captivo (2015) audited the Centro Hospitalar Lisboa Norte
  (CHLN) waiting list against SIGIC targets. Key findings:

  \begin{itemize}\small
    \item $\sim$\textbf{7{,}400} patients on the list at audit time.
    \item \textbf{16\%} had \emph{already} breached their tier's
      maximum-wait deadline.
    \item Average overrun among breached cases: \textbf{147 days}
      hospital-wide.
    \item Neurosurgery: average overrun of $\sim$\textbf{261 days} --
      far worse than the hospital-wide average.
  \end{itemize}
  \pause
  \begin{block}{Why this shapes this model's objective}
    A planner optimising only throughput can report excellent utilization
    while quietly reproducing this audit's failure mode -- scheduling
    routine cases and deferring overdue ones indefinitely, because a
    FIFO queue or an unweighted objective can't distinguish them.
    The penalty term $w_c$ in the objective exists specifically to
    make breach costs \emph{explicit, weighted, and dominant}.
  \end{block}
\end{frame}

\section{Assumptions \& Scope}

\begin{frame}{What the model covers -- and what it doesn't}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{Modelled:}
      \begin{itemize}\small
        \item Case$\to$room$\to$time$\to$surgeon assignment
        \item Room/service rosters (block schedule)
        \item Surgeon availability, daily \& weekly limits
        \item One shared imaging unit (C-arm) with capacity
        \item Recovery/ICU bed pool (multi-day stay)
        \item Priority-4 emergency add-on hard lock-in
        \item Pediatric-block carve-out rule
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{Deliberately excluded:}
      \begin{itemize}\small
        \item Stochastic durations
        \item Same-day disruptions
        \item Nurse/anaesthetist rostering
        \item Sequence-dependent OR cleaning
        \item Multi-week rolling horizon
        \item Emergency capacity reserve sizing
        \item Patient day-of-week preferences
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Key assumptions -- detail and justification}
  \small
  \begin{enumerate}
    \item \textbf{Deterministic durations.} One duration per case
      (e.g.\ historical median for that procedure). Real durations are
      stochastic and biased \emph{long}; the deterministic case is the
      standard tractable approximation (Denton et al., 2010). Stochastic
      extension requires two-stage SP -- top future-work item.
    \item \textbf{Turnover bucketed by duration} (15/25/40 min): proxy
      for the real 15--60 min range. Captures the duration-correlated
      component; not the sequence-dependent one (that's CP Optimizer,
      backup appendix).
    \item \textbf{Surgeons as binding staffing resource.} Nurses /
      anaesthetists assumed bundled with the rostered room -- valid when
      the surgeon's calendar is the actual bottleneck.
    \item \textbf{One occurrence per patient per week.} Conservative
      default (combining same-day procedures is service-specific clinical
      judgment, not assumed here).
    \item \textbf{Constant bed capacity across the week}, with explicit
      overflow charge for stays crossing the horizon -- no silent
      approximation.
    \item \textbf{Pediatric block} as a worked example of an ad hoc
      institutional rule: one eligibility predicate, no new variables.
    \item \textbf{Single week, single hospital, no same-day disruption}
      -- offline advance scheduling only.
  \end{enumerate}
\end{frame}

\section{Data \& Instances}

\begin{frame}{Data model}
  The data layer (\texttt{src/model/types.py}) is solver-agnostic --
  plain Python dataclasses, no solver imports.

  \begin{itemize}\small
    \item \textbf{\texttt{SurgicalCase}}: patient id, service, surgeon,
      priority $p_c$, days waited $\text{wl}_c$, operative time $t_c^{op}$,
      turnover $t_c^{clean}$, optional equipment, optional bed pool.
    \item \textbf{\texttt{Surgeon}}: daily limit $k_{hd}$, weekly
      limit $k_h$, per-day availability.
    \item \textbf{\texttt{OperatingRoom}}: per-day capacity
      $k_{dr}$, service roster, ambulatory-only flag.
    \item \textbf{\texttt{PlanningInstance}}: collects all the above
      plus policy parameters ($\text{wl}^{max}_p$, $\mu_p$, $\alpha$,
      equipment/bed capacities, pediatric-block rule).
  \end{itemize}
  \vspace{0.4em}
  Every solver reads \texttt{PlanningInstance} and returns
  \texttt{SolverResult} -- the solver adapter pattern that makes
  backends swappable.
\end{frame}

\section{The CP-SAT Model}

\begin{frame}{Sets and indices}
  \begin{center}\small
  \begin{tabular}{cl}
    \toprule
    Symbol & Meaning \\
    \midrule
    $c \in C$ & Surgical cases (one entry per patient-procedure pair) \\
    $d \in D$ & Planning days, $D=\{1,\dots,5\}$ (Mon--Fri) \\
    $r \in R$ & Operating rooms \\
    $h \in H$ & Surgeons \\
    $e \in E$ & Shared equipment types (one unit per type in the demo) \\
    $D_c \subseteq D$ & Eligible days for case $c$; $\{1\}$ if priority-4 \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.5em}
  \footnotesize Candidates $(c,d,r)$ are pre-filtered by eligibility
  (C4--C6) \emph{before} any variable is created -- this is the model's
  main variable-count reduction, not a constraint row.
\end{frame}

\begin{frame}{Parameters (I): time, capacity, priority}
  \small
  \begin{center}
  \begin{tabular}{cll}
    \toprule
    Symbol & Meaning & Default / source \\
    \midrule
    $t_c^{op}$ & Operative duration (min) & Historical median \\
    $t_c^{clean}$ & Turnover (min) & 15/25/40 by duration bucket \\
    $t_c^{tot}=t_c^{op}+t_c^{clean}$ & Total room occupation & --- \\
    $k_{dr}$ & Room $r$ open minutes on day $d$ & Hospital roster \\
    $k_{hd}$ & Surgeon daily op. limit (min) & 240 (demo) / 300 (med) \\
    $k_h$ & Surgeon weekly op. limit (min) & 960 (demo) / 1300 (med) \\
    $p_c\in\{1,2,3,4\}$ & Clinical priority tier & SIGIC triage \\
    $\text{wl}_c$ & Days waited (planning date) & Hospital data \\
    $\text{wl}^{max}_p$ & Max acceptable wait for tier $p$ & 270/60/15/3 (SIGIC) \\
    $dd_c=\text{wl}^{max}_{p_c}-\text{wl}_c$ & Slack to deadline ($<0$ = overdue) & Computed \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{frame}

\begin{frame}{Parameters (II): objective and shared resources}
  \small
  \begin{center}
  \begin{tabular}{cll}
    \toprule
    Symbol & Meaning & Default / source \\
    \midrule
    $\mu_p$ & Priority multiplier & $\text{wl}^{max}_1/\text{wl}^{max}_p$ (derived, see below) \\
    $w_c$ & Non-scheduling penalty & $\mu_{p_c}\cdot\text{PenaltyCurve}(dd_c)+1.2\max dd$ \\
    $\alpha>1$ & Urgency amplifier for overdue cases & 2.0 (day-ordering knob) \\
    $u_{ce}\in\{0,1\}$ & 1 if case $c$ needs equipment $e$ & Instance data \\
    $\kappa_{ed}$ & Simultaneous capacity of equip. $e$ on day $d$ & 1 (demo), 2 (med) \\
    $\rho(c)$ & Recovery bed pool required (``none'' if not needed) & Instance data \\
    $\text{los}_c$ & Length of stay in that pool (days) & Instance data \\
    $\beta_\rho$ & Bed count in pool $\rho$ (constant/week) & 2 (demo), 6 (med) \\
    $\pi^{ovf}$ & Penalty/day for bed stay past horizon & 50 (configurable) \\
    \bottomrule
  \end{tabular}
  \end{center}
  \footnotesize $\mu_p$: tier allowed only $1/12$ of P1's wait is
  12$\times$ more wait-sensitive: $\mu_2=270/60=4.5$,
  $\mu_3=270/15=18$, $\mu_4=270/3=90$.
\end{frame}

\begin{frame}{Decision variables -- presence and scheduling}
  For every $(c,d,r)$ surviving eligibility:
  \[\small
    \text{pr}_{cdr}\in\{0,1\}, \qquad
    \text{start}_{cdr},\,\text{end}_{cdr}\in[0,\,k_{dr}]
  \]
  For every case that is not priority-4:
  \[\small
    u_c\in\{0,1\}, \qquad
    \sum_{d\in D_c,\,r}\text{pr}_{cdr} + u_c = 1
  \]
  \begin{itemize}\small
    \item $\text{pr}_{cdr}=1$: case $c$ is scheduled on day $d$ in room $r$.
    \item $u_c=1$: case $c$ is \emph{not} scheduled this week and incurs
      penalty $w_c$.
    \item Priority-4 cases have no $u_c$: they are forced to be scheduled
      (constraint C2 handles this).
    \item Only $(c,d,r)$ triples passing C4--C6 eligibility get a
      variable -- the main variable-count reduction.
  \end{itemize}
\end{frame}

\begin{frame}{Decision variables -- two interval variables per candidate}
  \[\small
    \text{iv}_{cdr}=\texttt{NewOptionalIntervalVar}(\text{start}_{cdr},\;t_c^{tot},\;\text{end}_{cdr},\;\text{pr}_{cdr})
    \;\text{(ROOM)}
  \]
  \[\small
    \text{sgiv}_{cdr}=\texttt{NewOptionalIntervalVar}(\text{start}_{cdr},\;t_c^{op},\;\text{sgend}_{cdr},\;\text{pr}_{cdr})
    \;\text{(SURGEON)}
  \]
  Both share \texttt{start}; both are \emph{present} (constrain their
  resource) exactly when $\text{pr}_{cdr}=1$.
  \pause
  \begin{block}{Why two intervals, not one?}
    \small Surgeon H operates Case X, $t_c^{op}=70$, $t_c^{clean}=20$,
    starting at $t=0$. The \emph{room} must stay blocked until $t=90$.
    The \emph{surgeon} is free at $t=70$ and can scrub into Room 2 for
    Case Y while nursing staff finish cleaning Room 1.
    Using a single $t_c^{tot}$-sized interval for C8 would wrongly forbid
    that -- the surgeon would appear busy until $t=90$ in any room.
    Separating C7 (room, $t_c^{tot}$) from C8 (surgeon, $t_c^{op}$) on
    the same \texttt{start} resolves this without an explicit channeling
    constraint.
  \end{block}
\end{frame}

\begin{frame}{CP-SAT variables -- complete formal reference}
  \tiny
  \begin{center}
  \begin{tabular}{p{3.2cm}p{3cm}p{6.2cm}}
    \toprule
    \textbf{Variable} & \textbf{Domain} & \textbf{Meaning} \\
    \midrule
    $\text{pr}_{cdr}$
      & $\{0,1\}$
      & 1 if case $c$ is scheduled on day $d$ in room $r$; created
        only for eligible $(c,d,r)$ triples (C4--C6 pre-filter) \\[2pt]
    $\text{start}_{cdr}$
      & $[0,\,k_{dr}]$
      & Start time (min from room opening) when $\text{pr}_{cdr}=1$ \\[2pt]
    $\text{end}_{cdr}$
      & $[0,\,k_{dr}]$
      & Room-block end = $\text{start}_{cdr}+t_c^{tot}$ (room interval boundary) \\[2pt]
    $\text{iv}_{cdr}$
      & \texttt{OptionalIntervalVar}
      & \textbf{Room} occupancy interval: $[\text{start},\,\text{start}+t_c^{tot})$;
        active iff $\text{pr}_{cdr}=1$; feeds C7, C10, C11 \\[2pt]
    $\text{sgend}_{cdr}$
      & $[0,\,k_{dr}]$
      & Surgeon-end = $\text{start}_{cdr}+t_c^{op}$ (surgeon interval boundary) \\[2pt]
    $\text{sgiv}_{cdr}$
      & \texttt{OptionalIntervalVar}
      & \textbf{Surgeon} time interval: $[\text{start},\,\text{start}+t_c^{op})$;
        active iff $\text{pr}_{cdr}=1$; feeds C8 only.
        Shorter than $\text{iv}_{cdr}$ by $t_c^{clean}$ \\[2pt]
    $u_c$
      & $\{0,1\}$
      & 1 if case $c$ is \emph{not} scheduled (penalty $w_c$ applies);
        only for $p_c\ne4$.
        Enforced by $\sum_{d,r}\text{pr}_{cdr}+u_c=1$ \\[2pt]
    $\text{dayof}_c$
      & $\{0,\dots,n_{d}-1\}$
      & Index of the day $c$ is scheduled on; derived via
        \texttt{OnlyEnforceIf} from whichever $\text{pr}_{cdr}=1$.
        Only built for cases with $\rho(c)\ne\text{none}$ \\[2pt]
    $\text{bed}_c$
      & \texttt{OptionalIntervalVar}
      & ICU/recovery interval: $[\text{dayof}_c,\,\text{dayof}_c+\text{los}_c)$;
        present iff $1-u_c=1$; feeds C11 Cumulative \\[2pt]
    $\text{overflow}_c$
      & $[0,\,\text{los}_c]$
      & Days of bed stay past the horizon:
        $\max(0,\,\text{dayof}_c+\text{los}_c-n_d)$; adds Term 4 to objective \\
    \bottomrule
  \end{tabular}
  \end{center}
\end{frame}

\begin{frame}{Objective function -- complete formula}
  \[\small
    \min \;
    \underbrace{\sum_{\substack{c:\,dd_c\ge0\\d\in D_c,r}}
      [dd_c + d]\;\text{pr}_{cdr}}_{\text{Term 1 -- on-time, prefer earlier}}
    +\;
    \underbrace{\sum_{\substack{c:\,dd_c<0\\d\in D_c,r}}
      [dd_c + \alpha d]\;\text{pr}_{cdr}}_{\text{Term 2 -- overdue, urgency-weighted}}
  \]
  \[\small
    +\;
    \underbrace{\sum_{c:\,p_c\ne4} w_c\,u_c}_{\text{Term 3 -- non-scheduling penalty}}
    +\;
    \underbrace{\pi^{ovf}\!\!\sum_{c:\,\rho(c)\ne\text{none}}\!\!\text{overflow}_c}_{\text{Term 4 -- bed-overflow penalty}}
  \]
  \begin{itemize}\footnotesize
    \item Term 1: rewards scheduling any case, preferring earlier days.
    \item Term 2: for overdue cases, $\alpha>1$ makes the planner front-load them.
    \item Term 3: dominant term -- ensures the model schedules rather than drops.
    \item Term 4: penalises bed stays running past the week, making cost explicit.
  \end{itemize}
\end{frame}

\begin{frame}{Non-scheduling penalty $w_c$ and priority calibration}
  \[\small
    w_c = \mu_{p_c}\cdot\text{PenaltyCurve}(dd_c) + 1.2\cdot\max_{c'\in C}\,dd_{c'}
  \]
  \textbf{PenaltyCurve} (\texttt{src/model/penalty.py}): flat while
  the case has slack, escalates sharply on breach, keeps rising with
  backlog -- the shape SIGIC-style systems use to price breach
  increasingly over time.

  \vspace{0.4em}
  \textbf{Priority multipliers} $\mu_p$ -- \emph{derived} from SIGIC:
  \[\small
    \mu_p = \frac{\text{wl}^{max}_1}{\text{wl}^{max}_p}
    \;\Rightarrow\;
    \mu_2=\tfrac{270}{60}=4.5,\quad
    \mu_3=\tfrac{270}{15}=18,\quad
    \mu_4=\tfrac{270}{3}=90
  \]
  A tier allowed only $1/12$ of P1's wait is, by that \emph{same}
  policy, 12$\times$ more wait-sensitive.

  \vspace{0.3em}
  \textbf{Displacement margin 1.2}: the minimum needed is
  $1+\alpha n_{\text{days}}/\max dd_c \approx 1.037$;
  1.2 gives a safety buffer across instances with shorter max-$dd$.
\end{frame}

\begin{frame}{Constraints C1--C3: case disposition}
  \textbf{C1 -- at most one occurrence per patient per week:}
  \[\small
    \sum_{c:\,\text{patient}(c)=n}\sum_{d\in D_c,r}\text{pr}_{cdr}\le 1
    \qquad\forall n
  \]
  \textbf{C2 -- priority-4 locked to day 1:}
  \[\small
    \sum_{r}\text{pr}_{c,1,r}=1 \qquad \forall c:\;p_c=4
    \quad\text{(all other slots for $c$ forced to 0)}
  \]
  \textbf{C3 -- every other case is scheduled exactly once or
  penalised:}
  \[\small
    \sum_{d\in D_c,r}\text{pr}_{cdr} + u_c = 1 \qquad\forall c:\;p_c\ne4
  \]
  \footnotesize \textbf{C4--C6 (eligibility):} room-service roster,
  ambulatory-only flag, pediatric-block age rule -- all three are
  resolved during candidate generation; they never appear as a
  constraint row and add no overhead to the solver.
\end{frame}

\begin{frame}{Constraints C7--C9: room and surgeon}
  \textbf{C7 -- room, exact non-overlap:}
  \[\small
    \texttt{AddNoOverlap}\!\left(\left\{\text{iv}_{cdr}:\,d,r\text{ fixed}\right\}\right)
    \qquad\forall d\in D,\;r\in R
  \]
  \textbf{C8 -- surgeon, exact non-overlap on \emph{surgeon's} interval
  + daily cap:}
  \[\small
    \texttt{AddNoOverlap}\!\left(\left\{\text{sgiv}_{cdr}:\,\text{surgeon}(c)=h,\;
    d\text{ fixed}\right\}\right)\;\forall h,d
  \]
  \[\small
    \sum_{c:\,\text{surgeon}(c)=h}\sum_r t_c^{op}\,\text{pr}_{cdr}\le k_{hd}
    \qquad\forall h\in H,\;d\in D
  \]
  \textbf{C9 -- surgeon weekly limit:}
  \[\small
    \sum_{c:\,\text{surgeon}(c)=h}\sum_{d,r} t_c^{op}\,\text{pr}_{cdr}\le k_h
    \qquad\forall h\in H
  \]
  \footnotesize NoOverlap bounds concurrency; C8/C9 cap total hours --
  both are needed because they constrain different things.
\end{frame}

\begin{frame}{Constraints C10--C11: equipment and beds}
  \textbf{C10 -- shared equipment, exact time-overlap check:}
  \[\small
    \texttt{AddCumulative}\!\left(
      \left\{\text{iv}_{cdr}:\,u_{ce}=1,\;d\text{ fixed}\right\},\;
      \mathbf{1},\;\kappa_{ed}
    \right)
    \qquad\forall e\in E,\;d\in D
  \]
  \textbf{C11 -- recovery/ICU beds, with horizon-overflow charge:}
  \[\small
    \text{dayof}_c = d_{\text{idx}}
    \;\text{(via \texttt{OnlyEnforceIf} when $\text{pr}_{cdr}=1$)}
  \]
  \[\small
    \text{bed}_c=\texttt{NewOptionalIntervalVar}
    (\text{dayof}_c,\;\text{los}_c,\;\text{dayof}_c+\text{los}_c,\;1-u_c)
  \]
  \[\small
    \texttt{AddCumulative}\!\left(\left\{\text{bed}_c:\;\rho(c)=\rho\right\},\;\mathbf{1},\;\beta_\rho\right)
    \qquad\forall\rho
  \]
  \[\small
    \text{overflow}_c=\max(0,\;\text{dayof}_c+\text{los}_c-n_{\text{days}})
  \]
  \footnotesize C11 needs a real ``day of surgery'' value -- the
  concrete reason this model is interval-based. A day-bucket MIP has
  no such variable (see backup appendix).
\end{frame}

\begin{frame}{CP-SAT search strategy}
  \begin{itemize}\small
    \item \textbf{Parallel portfolio} -- several complete-search workers
      with different heuristics, plus Large Neighbourhood Search workers
      improving an incumbent, sharing learned clauses via a common SAT
      core (Perron \& Furnon, Google OR-Tools docs).
    \item \textbf{No custom branching strategy} -- OR-Tools' guidance is
      that the default portfolio outperforms hand-tuned single strategies
      unless you have structural insight not already exposed through
      \texttt{NoOverlap}/\texttt{Cumulative}. This model exposes that
      structure fully.
    \item \textbf{Workers} capped at \texttt{os.cpu\_count()} -- over-
      subscribing past physical cores adds contention, not search
      diversity.
    \item \textbf{Gap target} $=1\%$ by default; solver stops when the
      best bound is within 1\% of the current incumbent.
    \item \textbf{``Optimal'' means within the configured gap}, not a
      literal 0\% gap -- always printed and never assumed zero.
  \end{itemize}
\end{frame}

\section{Why CP-SAT, Not a MIP}

\begin{frame}{The core argument}
  \begin{block}{Key distinction}
    ``Total minutes of equipment use today $\le$ capacity'' certifies
    that a set of durations \emph{fits}. It does \emph{not} certify
    they can be placed without \emph{colliding}. For one room the two
    coincide (non-overlapping durations can always be packed
    sequentially). They \textbf{stop coinciding} when a resource is
    shared across multiple rooms.
  \end{block}
  \begin{columns}
    \begin{column}{0.5\textwidth}\small
      \textbf{MIP (day-bucket):} count C-arm cases per day $\le 1$.\
      Legal placements forbidden: 2 non-overlapping C-arm cases
      on the same day fail this check even if they never overlap.
    \end{column}
    \begin{column}{0.5\textwidth}\small
      \textbf{CP-SAT:} \texttt{AddCumulative} checks literal time
      overlap. Two sequential C-arm cases on the same day -- never
      concurrent -- pass the check. Larger feasible region, better
      possible schedules.
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Advantages and disadvantages}
  \small
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{CP-SAT / advantages:}
      \begin{itemize}
        \item \texttt{NoOverlap}/\texttt{Cumulative} with poly-time propagation (Vil\'im 2004; Schutt et al. 2009)
        \item Exact intra-day clock times modelled natively
        \item Bed constraint C11 expressible at all
        \item Parallel portfolio search (LNS + complete search)
        \item No big-M constants, no LP-relaxation weakening
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{MILP / advantages:}
      \begin{itemize}
        \item Strong LP relaxation bounding via cutting-plane theory
        \item Mature commercial solvers (Gurobi, CPLEX)
        \item Proven MIP gap closed very fast on small-to-medium instances
        \item Own gap closes more tightly on a smaller feasible region
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.4em}
  \begin{columns}
    \begin{column}{0.5\textwidth}
      \textbf{CP-SAT / disadvantages:}
      \begin{itemize}
        \item Own gap can be looser (larger search space)
        \item Less interpretable as a ``classical'' LP
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}
      \textbf{MILP / disadvantages:}
      \begin{itemize}
        \item Quadratic variable growth for disjunctions
        \item Big-M weakens LP relaxation
        \item Cannot express C11 (multi-day bed interval)
        \item Day-count equipment cap \emph{logically incorrect} for shared resources
      \end{itemize}
    \end{column}
  \end{columns}
  \miplink
\end{frame}

\begin{frame}{When would MILP be the right choice instead?}
  \small Even within this problem class, a MILP formulation is the more
  natural or practical choice in specific situations:
  \vspace{0.4em}
  \begin{itemize}
    \item \textbf{Day-assignment only, no clock times needed.} When the
      deliverable is ``case $c$ runs on day $d$ in room $r$'' without
      an exact start time, the day-bucket MIP naturally captures this;
      interval variables would be over-engineering.
    \item \textbf{No shared resources across rooms.} If each piece of
      equipment is dedicated to one room and surgeons don't move between
      rooms, the day-count constraint is sufficient -- the pathological
      case that drives the CP-SAT choice doesn't arise.
    \item \textbf{No multi-day downstream constraint (C11).} When recovery/
      ICU beds are not modelled, one key structural reason for intervals
      disappears.
    \item \textbf{Small instance, LP relaxation is tight.} A commercial
      solver (Gurobi, CPLEX) can close the MIP gap in seconds with a
      provably optimal LP-relaxation certificate -- faster convergence
      than CP-SAT on instances where the feasible region is small.
    \item \textbf{Need LP dual variables.} Shadow prices from the LP
      relaxation support sensitivity analysis and what-if reporting --
      something CP-SAT doesn't expose directly.
    \item \textbf{Team expertise is in MILP.} When the implementation
      team is more fluent in LP/MIP modelling than CP, the modelling
      risk of incorrectly encoding constraints is lower with the tool
      they know better.
  \end{itemize}
\end{frame}

\begin{frame}{Conclusions: what the experiments show}

      more case. The reason is always the same: a larger, correct feasible
      region.
    \item \textbf{The gap metric does not mean what it seems.}
      CP-SAT's 6.62\% gap and CBC's 0.56\% gap are measured against
      \emph{different} feasible-region bounds. The right comparison is the
      absolute objective value; CP-SAT wins by a wide margin.
    \item \textbf{Shared equipment modeling is the decisive structural factor.}
      \texttt{AddCumulative} checks actual time overlap; the MIP's day-count
      cap forbids two sequential C-arm cases on the same day even when they
      never overlap -- a logically incorrect constraint.
    \item \textbf{C11 (recovery/ICU beds) cannot be expressed in a day-bucket
      MIP.} This alone is a structural justification for interval-based CP,
      independent of performance.
    \item \textbf{The two-interval design enables real OR practice.} A surgeon
      free at $t=70$ can enter a second room while Room 1 cleans until $t=90$.
      A naive single-interval model incorrectly forbids this.
    \item \textbf{The demo instance solves in $\sim$0.1 s to proven zero-gap
      optimality} -- entirely practical even in interactive decision support.
    \item \textbf{At a 2-minute budget, CP-SAT is already more valuable than
      CBC.} More cases scheduled, lower objective, same wall-clock time.
  \end{enumerate}

\section{Backup: Comparison MIP (Appendix Model)}

\begin{frame}{MIP -- formulation overview}
  \hypertarget{mip1}{}
  The comparison MIP (FORMULATION.md, Appendix A) is implemented in
  \texttt{milp\_baseline\_solver.py} and runs via
  \texttt{-{}-solver milp-cbc} (bundled), \texttt{milp-gurobi}, or
  \texttt{milp-cplex} (commercial license required).

  \textbf{Same as CP-SAT}: sets, eligibility pre-filter (C4--C6),
  priority tiers, $w_c$ from the same \texttt{penalty.py}.

  \textbf{Key differences}:
  \begin{itemize}\small
    \item No intra-day clock times -- decisions are at \emph{day+room}
      granularity only.
    \item C7 and C10 are linear capacity sums, not exact non-overlap.
    \item C11 (recovery beds) is \emph{not expressible} -- a day-bucket
      variable has no ``day of surgery'' value for a multi-day stay.
    \item No Term 4 in the objective (overflow penalty) -- C11 doesn't
      exist.
  \end{itemize}
  \bk
\end{frame}

\begin{frame}{MIP -- variables, complete reference}
  \footnotesize
  \begin{center}
  \begin{tabular}{p{2.6cm}p{2.8cm}p{6.4cm}}
    \toprule
    \textbf{Variable} & \textbf{Domain} & \textbf{Meaning} \\
    \midrule
    $x_{cdr}$ & $\{0,1\}$ &
      1 if case $c$ is scheduled on day $d$ in room $r$.
      Only created for eligible $(c,d,r)$ triples (same C4--C6 pre-filter
      as CP-SAT). Determines day and room but \emph{not} exact time. \\[3pt]
    $z_c$ & $[0,1]$ &
      1 if case $c$ is \emph{not} scheduled this week (non-priority-4 only).
      Declared continuous -- C3 forces it to $\{0,1\}$ at the optimum
      because the objective penalises fractional non-scheduling; declaring
      it binary adds no useful LP-relaxation tightening. \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.4em}
  \textbf{Objective} (same three-term structure as CP-SAT, no Term 4):
  \[\footnotesize
    \min\;\sum_{c:\,dd_c\ge0}\sum_{d,r}[dd_c+d]\,x_{cdr}
    +\sum_{c:\,dd_c<0}\sum_{d,r}[dd_c+\alpha d]\,x_{cdr}
    +\sum_{c:\,p_c\ne4}w_c\,z_c
  \]
  Same \texttt{penalty.py} computes $w_c$ -- no separate implementation
  per solver. No Term 4 because C11 (recovery beds) is not expressible
  in this formulation.
  \bk
\end{frame}

\begin{frame}{MIP -- constraints and what's missing}
  \textbf{C7 (room, aggregate sum):}
  \[\small
    \sum_{c\in C}t_c^{tot}\,x_{cdr}\le k_{dr}\qquad\forall d,r
  \]
  For a single room: equivalent to exact non-overlap. For a \emph{shared
  resource across rooms}: incorrectly counts all equipment-uses as if
  they overlap.

  \textbf{C10 (equipment, daily headcount):}
  \[\small
    \sum_{c:\,u_{ce}=1}\sum_r x_{cdr}\le\kappa_{ed}\qquad\forall e,d
  \]
  Counts \emph{how many} equipment-$e$ cases land on a day, not whether
  they overlap in clock time -- the main source of the 5.56\% objective
  gap vs. CP-SAT at scale.

  \textbf{No C11:} ``$\text{bed}_c$ starts on whichever day $c$
  is scheduled'' cannot be written -- no variable in a day-bucket model
  holds ``day of surgery'' as a \emph{value}.
  \bk
\end{frame}

\begin{frame}{CP Optimizer (optional, appendix)}
  \small Kept for one reason: \textbf{sequence-dependent room turnover}
  -- a transition matrix between adjacent cases in a room via
  \texttt{sequence\_var + no\_overlap(transition\_matrix)}.
  \begin{center}
  \begin{tabular}{lcc}
    \toprule
    & Same service & Different service \\
    \midrule
    CP-SAT (duration-bucketed) & case's own $t_c^{clean}$ & same \\
    CP Optimizer & 15 min & 35 min \\
    \bottomrule
  \end{tabular}
  \end{center}
  Demo instance: identical to CP-SAT (155.0, 20/20, Optimal) -- confirmed
  that same-service turnover gaps equal exactly 15 min.

  At 200 cases, own gap closes slower than CP-SAT (search-tuning
  difference -- no warm start applied; not a modelling weakness).
  \bk
\end{frame}
