\section{Problema e Caso de Estudo}

\begin{frame}{O problema de planeamento}
  \textbf{Um hospital, uma semana. Uma lista de espera cir\'urgica sempre
  maior do que a capacidade semanal dos blocos operat\'orios.}
  \begin{itemize}\small
    \item \textbf{Decis\~ao por caso:} que bloco, que dia, que hora de
      in\'icio, que cirurgi\~ao -- ou adiar para a semana seguinte.
    \item \textbf{Exequibilidade:} blocos, cirurgi\~oes, equipamento
      partilhado e camas de recupera\c{c}\~ao s\~ao todos finitos e
      partilhados.
    \item \textbf{Equidade:} a urg\^encia cl\'inica, e n\~ao a ordem
      de chegada (FIFO), deve determinar a prioridade -- um doente
      em incumprimento n\~ao pode esperar indefinidamente atr\'as de
      um caso de rotina.
    \item \textbf{Custo:} o tempo de bloco operat\'orio custa
      $\sim\$30$--$\$40$/min uma vez amortizados pessoal e
      sobrecustos (Macario, 2010). Um minuto vazio hoje n\~ao pode
      ser poupado para amanh\~a.
    \item \textbf{Escopo:} \emph{planeamento antecipado} -- atribuir
      casos a dias e blocos antes do in\'icio da semana
      (Cardoen et al., 2010) -- distinto do sequenciamento intra-dia
      ou da gest\~ao de perturba\c{c}\~oes em tempo real.
  \end{itemize}
\end{frame}

\begin{frame}{Caso de estudo: o sistema SIGIC em Portugal}
  \textbf{SIGIC} (\emph{Sistema Integrado de Gest\~ao de Inscritos
  para Cirurgia}) -- sistema nacional de gest\~ao da lista de espera
  cir\'urgica em Portugal, criado pela \textbf{Portaria
  n.\textsuperscript{o} 45/2008} (Di\'ario da Rep\'ublica).

  \vspace{0.4em}
  Cada caso electivo \'{e} classificado numa de \textbf{quatro
  prioridades cl\'inicas}, cada uma com uma \textbf{espera m\'axima
  admiss\'ivel} legislada:

  \vspace{0.3em}
  \begin{center}\small
  \begin{tabular}{clcc}
    \toprule
    Prioridade & Significado cl\'inico & Espera m\'ax. & Par\^ametro do modelo \\
    \midrule
    1 & Electivo de rotina & 270 dias & $\text{wl}^{max}_1 = 270$ \\
    2 & Urg\^encia elevada & 60 dias  & $\text{wl}^{max}_2 = 60$  \\
    3 & Urgente            & 15 dias  & $\text{wl}^{max}_3 = 15$  \\
    4 & Emergente (operar esta semana) & 3 dias & $\text{wl}^{max}_4 = 3$ \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3em}
  \footnotesize Estes quatro n\'umeros s\~ao os par\^ametros
  $\text{wl}^{max}_p$ do modelo -- retirados directamente de uma
  pol\'itica real, legislada e auditada anualmente. A mesma estrutura
  aparece nos objectivos RTT do NHS brit\^anico e nos benchmarks
  provinciais canadianos.
\end{frame}

\begin{frame}{Caso de estudo: a auditoria CHLN de 2016}
  Marques \& Captivo (2015) auditaram a lista de espera do
  \textbf{Centro Hospitalar Lisboa Norte (CHLN)} face aos objectivos
  SIGIC:
  \begin{itemize}\small
    \item $\sim$\textbf{7{,}400} doentes na lista de espera no momento
      da auditoria.
    \item \textbf{16\%} j\'a tinham \emph{ultrapassado} o prazo m\'aximo
      da sua prioridade.
    \item M\'edia de incumprimento nos casos em viola\c{c}\~ao:
      \textbf{147 dias} a n\'ivel hospitalar.
    \item Pior servi\c{c}o -- Neurocirurgia: m\'edia de
      $\sim$\textbf{261 dias} de incumprimento nos casos violados.
  \end{itemize}
  \pause
  \begin{block}{Porque estes n\'umeros moldam o objectivo do modelo}
    Um planeador que maximize apenas o n\'umero de cirurgias pode
    reportar excelente utiliza\c{c}\~ao, reproduzindo silenciosamente
    esta falha: agendando casos de rotina e adiando indefinidamente os
    mais urgentes, porque uma fila FIFO ou um objectivo n\~ao ponderado
    n\~ao consegue distingui-los.
    A \textbf{penaliza\c{c}\~ao de n\~ao-agendamento $w_c$} existe
    precisamente para tornar os custos de viola\c{c}\~ao
    \emph{expl\'icitos, ponderados pela prioridade e dominantes}
    no objectivo.
  \end{block}
\end{frame}

\section{Modelo CP-SAT}

\begin{frame}{O que o modelo cobre -- e o que exclui}
  \begin{columns}
    \begin{column}{0.48\textwidth}
      \textbf{Modelado:}
      \begin{itemize}\footnotesize
        \item Atribui\c{c}\~ao caso $\to$ bloco $\to$ hora $\to$ cirurgi\~ao
        \item Ros\'arios de blocos por servi\c{c}o
        \item Disponibilidade do cirurgi\~ao, limites di\'arios e semanais
        \item Uma unidade de imagiologia partilhada (intensificador C)
        \item Camas UCI / recupera\c{c}\~ao (interna\c{c}\~ao multi-dia)
        \item Bloqueio obrigat\'orio ao dia 1 para prioridade 4
        \item Regra de bloco pedi\'atrico
      \end{itemize}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Deliberadamente exclu\'ido:}
      \begin{itemize}\footnotesize
        \item Dura\c{c}\~oes estoc\'asticas
        \item Perturba\c{c}\~oes intra-dia / urg\^encias n\~ao programadas
        \item Ros\'ario de enfermeiros / anestesistas
        \item Limpeza dependente da sequ\^encia
        \item Horizonte multi-semana
        \item Dimensionamento de reserva de capacidade para urg\^encias
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{0.5em}
  \textbf{Pressupostos-chave (justifica\c{c}\~ao completa: FORMULATION.md \S2):}
  \begin{itemize}\footnotesize
    \item Dura\c{c}\~oes determin\'isticas (mediana hist\'orica por procedimento).
    \item Limpeza categorizada por dura\c{c}\~ao do caso: 15 / 25 / 40 min.
    \item Cirurgi\~ao como recurso de pessoal limitante (enfermeiros
      assumidos com o bloco ros\'ariado).
    \item Capacidade de camas constante; encargo expl\'icito por excesso
      de interna\c{c}\~ao al\'em do horizonte.
  \end{itemize}
\end{frame}

\begin{frame}{Porqu\^e CP-SAT e n\~ao um MIP?}
  \begin{block}{Argumento central -- contar $\neq$ n\~ao-sobreposi\c{c}\~ao
  em recursos partilhados}
    ``Total de minutos de uso do equipamento hoje $\le$ capacidade''
    certifica que as dura\c{c}\~oes \emph{cabem}. N\~ao certifica
    que n\~ao \emph{colidem no tempo}. Para um \'unico bloco os dois
    coincidem; para um recurso partilhado entre v\'arios blocos
    deixam de coincidir.
  \end{block}
  \vspace{0.2em}
  \begin{columns}
    \begin{column}{0.48\textwidth}\footnotesize
      \textbf{MIP (balde de dia):} conta usos do intensificador C
      por dia $\le1$. Dois casos sequenciais no mesmo dia
      \emph{falham} esta verifica\c{c}\~ao mesmo que nunca
      se sobreponham $\Rightarrow$ calend\'arios vi\'aveis s\~ao
      proibidos.
    \end{column}
    \begin{column}{0.48\textwidth}\footnotesize
      \textbf{CP-SAT:} \texttt{AddCumulative} verifica sobreposi\c{c}\~ao
      temporal real. Dois casos sequenciais passam se nunca
      correram em simult\^aneo $\Rightarrow$ regi\~ao vi\'avel
      maior e correcta $\Rightarrow$ melhores calend\'arios.
    \end{column}
  \end{columns}
  \vspace{0.4em}
  \textbf{Raz\~ao estrutural -- C11 (camas UCI/recupera\c{c}\~ao):}
  a restri\c{c}\~ao ``$\text{bed}_c$ inicia no dia em que $c$ \'e
  operado'' requer uma vari\'avel que contenha o ``dia de cirurgia''
  como \emph{valor concreto}. Num MIP de balde de dia n\~ao existe
  essa vari\'avel -- C11 n\~ao pode ser expressa nessa formula\c{c}\~ao,
  independentemente do desempenho do solver.
  \miplink
\end{frame}

\begin{frame}{Conjuntos e par\^ametros}
  \begin{columns}
    \begin{column}{0.44\textwidth}
      \textbf{Conjuntos e \'indices}\vspace{0.2em}
      \begin{center}\footnotesize
      \begin{tabular}{cl}
        \toprule
        S\'imbolo & Significado \\
        \midrule
        $c\in C$ & Casos cir\'urgicos \\
        $d\in D$ & Dias $\{1,\dots,5\}$ \\
        $r\in R$ & Blocos operat\'orios \\
        $h\in H$ & Cirurgi\~oes \\
        $e\in E$ & Tipos de equipamento partilhado \\
        $D_c\subseteq D$ & Dias eleg\'iveis para $c$ \\
        \bottomrule
      \end{tabular}
      \end{center}
    \end{column}
    \begin{column}{0.54\textwidth}
      \textbf{Par\^ametros-chave}\vspace{0.2em}
      \begin{center}\footnotesize
      \begin{tabular}{cl}
        \toprule
        S\'imbolo & Significado \\
        \midrule
        $t_c^{op},t_c^{clean},t_c^{tot}$ & Dura\c{c}\~ao, limpeza, total \\
        $k_{dr}$, $k_{hd}$, $k_h$ & Cap. bloco / cirurgi\~ao di\'aria / semanal \\
        $p_c\in\{1..4\}$ & Prioridade SIGIC \\
        $\text{wl}_c$ & Dias em espera na data de planeamento \\
        $\text{wl}^{max}_p$ & Espera m\'ax. para tier $p$: 270/60/15/3 \\
        $dd_c = \text{wl}^{max}_{p_c}-\text{wl}_c$ & Folga ($<0$ = em viola\c{c}\~ao) \\
        $\mu_p = \text{wl}^{max}_1/\text{wl}^{max}_p$ & Multiplicador de prioridade \\
        $\alpha>1$ & Amplificador de urg\^encia para casos em viola\c{c}\~ao \\
        $u_{ce},\kappa_{ed}$ & Necessidade de equip., capacidade di\'aria \\
        $\rho(c),\text{los}_c,\beta_\rho$ & Piscina de camas, interna\c{c}\~ao, capacidade \\
        $\pi^{ovf}$ & Encargo por dia de excesso de interna\c{c}\~ao \\
        \bottomrule
      \end{tabular}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.3em}
  \footnotesize\textbf{Multiplicadores de prioridade derivados do SIGIC:}
  $\mu_2=270/60=4.5$,\; $\mu_3=270/15=18$,\; $\mu_4=270/3=90$.
  Um tier cuja pol\'itica permite apenas $1/18$ da espera de P1
  \'e, por essa mesma pol\'itica, $18\times$ mais sens\'ivel ao tempo.
\end{frame}

\begin{frame}{Vari\'aveis de decis\~ao}
  \textbf{Presen\c{c}a e agendamento} -- para cada $(c,d,r)$ eleg\'ivel:
  \[\small
    \text{pr}_{cdr}\in\{0,1\},\quad
    \text{start}_{cdr},\;\text{end}_{cdr}\in[0,k_{dr}]
  \]
  Para cada caso de prioridade n\~ao-4: $u_c\in\{0,1\}$,\quad
  $\sum_{d,r}\text{pr}_{cdr}+u_c=1$

  \vspace{0.3em}
  \textbf{Dois intervalos por candidato -- a escolha de design fundamental:}
  \[\small
    \text{iv}_{cdr}=\texttt{NewOptionalIntervalVar}(\text{start},\;t_c^{tot},\;\text{end},\;\text{pr}_{cdr})
    \quad\textbf{[BLOCO]}
  \]
  \[\small
    \text{sgiv}_{cdr}=\texttt{NewOptionalIntervalVar}(\text{start},\;t_c^{op},\;\text{sgend},\;\text{pr}_{cdr})
    \quad\textbf{[CIRURGI\~AO]}
  \]
  \begin{block}{Porqu\^e dois intervalos?}
    \small O cirurgi\~ao H opera o Caso X: $t^{op}=70$, $t^{clean}=20$,
    in\'icio em $t=0$. O \emph{bloco} fica ocupado at\'e $t=90$.
    O \emph{cirurgi\~ao} fica livre em $t=70$ e pode entrar no Bloco 2
    para o Caso Y enquanto o pessoal de enfermagem conclui a limpeza
    do Bloco 1. Um \'unico intervalo de tamanho $t^{tot}$ para C8
    bloquearia incorrectamente o cirurgi\~ao at\'e $t=90$ em qualquer
    bloco. Dar a C7 (bloco, $t^{tot}$) e a C8 (cirurgi\~ao, $t^{op}$)
    o mesmo \texttt{start} resolve isto sem restri\c{c}\~oes de
    liga\c{c}\~ao adicionais.
  \end{block}
\end{frame}

\begin{frame}{Fun\c{c}\~ao objectivo}
  \[\small
    \min\;
    \underbrace{\sum_{\substack{c:\,dd_c\ge0\\d\in D_c,r}}
      [dd_c+d]\,\text{pr}_{cdr}}_{\text{Termo 1 -- em prazo, mais cedo preferido}}
    +\;
    \underbrace{\sum_{\substack{c:\,dd_c<0\\d\in D_c,r}}
      [dd_c+\alpha d]\,\text{pr}_{cdr}}_{\text{Termo 2 -- em viola\c{c}\~ao, antecipados}}
    +\;
    \underbrace{\sum_{c:\,p_c\ne4} w_c\,u_c}_{\text{Termo 3 -- penaliza\c{c}\~ao}}
    +\;
    \underbrace{\pi^{ovf}\!\!\sum_{c:\,\rho(c)\ne\text{none}}\!\!\text{overflow}_c}_{\text{Termo 4 -- excesso de camas}}
  \]
  \[\small
    w_c = \mu_{p_c}\cdot\text{PenaltyCurve}(dd_c) + 1.2\cdot\max_{c'\in C}dd_{c'}
  \]
  \begin{itemize}\footnotesize
    \item \textbf{Termo 1:} agenda casos em prazo, preferindo dias mais cedo.
    \item \textbf{Termo 2:} para casos em viola\c{c}\~ao, $\alpha>1$
      antecipa-os para limitar o atraso adicional.
    \item \textbf{Termo 3 (dominante):} $w_c$ \'e calibrado para que o modelo
      prefira \emph{sempre} agendar a descartar; a penaliza\c{c}\~ao escala
      abruptamente ap\'os a viola\c{c}\~ao do prazo SIGIC.
    \item \textbf{Termo 4:} torna o custo de excesso de interna\c{c}\~ao
      expl\'icito; for\c{c}a o modelo a contabilizar estadias al\'em
      do horizonte de planeamento.
  \end{itemize}
\end{frame}

\begin{frame}{Restri\c{c}\~oes C1--C6: disposi\c{c}\~ao e elegibilidade}
  \textbf{C1 -- no m\'aximo uma ocorr\^encia por doente por semana:}
  \[\small
    \sum_{c:\,\text{doente}(c)=n}\!\sum_{d\in D_c,r}\text{pr}_{cdr}\le1
    \qquad\forall n
  \]
  \textbf{C2 -- casos de prioridade 4 bloqueados ao dia 1:}
  \[\small
    \sum_r\text{pr}_{c,1,r}=1
    \qquad\forall c:\;p_c=4
    \quad\text{(todos os outros slots for\c{c}ados a 0)}
  \]
  \textbf{C3 -- cada outro caso \'e agendado exactamente uma vez ou penalizado:}
  \[\small
    \sum_{d\in D_c,r}\text{pr}_{cdr}+u_c=1
    \qquad\forall c:\;p_c\ne4
  \]
  \textbf{C4--C6 -- elegibilidade (resolvida em tempo de constru\c{c}\~ao):}
  \begin{itemize}\footnotesize
    \item \textbf{C4:} correspond\^encia bloco-servi\c{c}o (ex.: casos VASC
      s\'o em blocos VASC).
    \item \textbf{C5:} blocos ambulat\'orios rejeitam casos com necessidade
      de interna\c{c}\~ao.
    \item \textbf{C6:} regra do bloco pedi\'atrico -- blocos ENT \`a
      sexta-feira s\'o para idade $\le8$ anos.
  \end{itemize}
  \footnotesize A elegibilidade corre uma \'unica vez. Nenhuma vari\'avel
  $\text{pr}_{cdr}$ \'e criada para um trio inelig\'ivel -- principal
  redu\c{c}\~ao do n\'umero de vari\'aveis.
\end{frame}

\begin{frame}{Restri\c{c}\~oes C7--C11: tempo, recursos e camas}
  \textbf{C7 -- n\~ao-sobreposi\c{c}\~ao de blocos (exacta):}\quad
  $\texttt{AddNoOverlap}(\{\text{iv}_{cdr}:\;d,r\text{ fixos}\})\;\forall d,r$

  \smallskip
  \textbf{C8 -- n\~ao-sobreposi\c{c}\~ao do cirurgi\~ao (no seu intervalo pr\'oprio):}
  \quad$\texttt{AddNoOverlap}(\{\text{sgiv}_{cdr}:\;\text{cir}(c)=h,\;d\text{ fixo}\})\;\forall h,d$

  \smallskip
  \textbf{C9 -- capacidade do cirurgi\~ao (di\'aria e semanal):}
  \[\small
    \sum_{c:\,\text{cir}(c)=h}\!\sum_r t_c^{op}\,\text{pr}_{cdr}\le k_{hd}\;\forall h,d
    \qquad
    \sum_{c,d,r} t_c^{op}\,\text{pr}_{cdr}\le k_h\;\forall h
  \]

  \textbf{C10 -- equipamento partilhado, verifica\c{c}\~ao exacta de sobreposi\c{c}\~ao:}
  \quad$\texttt{AddCumulative}(\{\text{iv}_{cdr}:\;u_{ce}=1,\;d\},\;\mathbf{1},\;\kappa_{ed})\;\forall e,d$

  \textbf{C11 -- camas UCI/recupera\c{c}\~ao (multi-dia, com excesso no horizonte):}
  \begin{itemize}\footnotesize
    \item $\text{dayof}_c\leftarrow d_\text{idx}$
      via \texttt{OnlyEnforceIf}($\text{pr}_{cdr}=1$).
    \item $\text{bed}_c=\texttt{NewOptionalIntervalVar}
      (\text{dayof}_c,\;\text{los}_c,\;\text{dayof}_c+\text{los}_c,\;1-u_c)$
    \item $\texttt{AddCumulative}(\{\text{bed}_c:\;\rho(c)=\rho\},\;\mathbf{1},\;\beta_\rho)
      \;\forall\rho$
    \item $\text{overflow}_c=\max(0,\;\text{dayof}_c+\text{los}_c-n_d)$
  \end{itemize}
  \footnotesize C10 \'e a restri\c{c}\~ao ``correcta do intensificador C''
  -- sobreposi\c{c}\~ao temporal real, n\~ao contagem di\'aria.
  C11 exige um valor concreto de ``dia de cirurgia'':
  estruturalmente imposs\'ivel num MIP de balde de dia.
\end{frame}

  \vspace{0.6em}

\section{Conclus\~oes}

\begin{frame}{Conclus\~oes}
  \begin{enumerate}\small
    \item \textbf{O CP-SAT encontra calend\'arios estritamente melhores.}
      Demo: 155.0 vs.\ 157.0 (\'optimo MIP provado). M\'edia: 69{,}956
      vs.\ 74{,}074 (\'optimo MIP provado) -- 5.56\% melhor, mais um
      caso agendado. A causa \'e sempre a mesma: regi\~ao vi\'avel
      maior e correcta.
    \item \textbf{As percentagens de desfasamento enganam sem contexto.}
      Os 6.62\% do CP-SAT e os 0.56\% do CBC s\~ao medidos face a
      fronteiras de \emph{regi\~oes vi\'aveis diferentes}. A compara\c{c}\~ao
      correcta \'e o valor absoluto do objectivo -- o CP-SAT vence.
    \item \textbf{A modela\c{c}\~ao do equipamento partilhado \'e o factor
      estrutural decisivo.} O \texttt{AddCumulative} verifica
      sobreposi\c{c}\~ao temporal real; a contagem di\'aria do MIP proibe
      incorrectamente usos n\~ao-sobrepostos do equipamento no mesmo dia.
    \item \textbf{C11 n\~ao pode ser expressa num MIP de balde de dia.}
      Nenhuma vari\'avel nessa formula\c{c}\~ao guarda o ``dia de
      cirurgia'' como valor -- argumento estrutural, independente do
      desempenho.
    \item \textbf{O design de dois intervalos reflecte a pr\'atica real do
      bloco.} O cirurgi\~ao livre em $t=70$ pode entrar no Bloco 2
      enquanto o Bloco 1 ainda est\'a a ser limpo at\'e $t=90$.
    \item \textbf{Inst\^ancia demo: \'optimo provado em $\sim$0.1 s.}
      Inteiramente pr\'atico para suporte a decis\~ao interactivo.
    \item \textbf{Com or\c{c}amento de 2 minutos, o CP-SAT j\'a domina o CBC}
      tanto no valor do objectivo como no n\'umero de casos agendados.
  \end{enumerate}
\end{frame}

\begin{frame}{Estrat\'egia de pesquisa do CP-SAT}
  \begin{itemize}\small
    \item \textbf{Port\'ef\'olio paralelo:} v\'arios workers de pesquisa
      completa com diferentes heur\'isticas de ordena\c{c}\~ao de
      vari\'aveis, mais workers LNS a melhorar uma solu\c{c}\~ao
      incumbente, todos partilhando cl\'ausulas aprendidas via um
      n\'ucleo SAT comum (Perron \& Furnon, Google OR-Tools).
    \item \textbf{Sem estrat\'egia de ramifica\c{c}\~ao personalizada:}
      as orienta\c{c}\~oes do OR-Tools indicam que o port\'ef\'olio
      predefinido supera estrat\'egias \'unicas ajustadas manualmente,
      excepto se existir conhecimento estrutural n\~ao j\'a exposto
      atrav\'es de \texttt{NoOverlap}/\texttt{Cumulative}.
    \item \textbf{Workers} limitados a \texttt{os.cpu\_count()} --
      exceder o n\'umero de n\'ucleos f\'isicos adiciona
      conten\c{c}\~ao, n\~ao diversidade de pesquisa.
    \item \textbf{Objectivo de desfasamento} $=1\%$ por defeito;
      o solver p\'ara quando a melhor fronteira est\'a a menos de 1\%
      do incumbente.
    \item \textbf{``\'Optimo'' significa dentro do desfasamento
      configurado}, n\~ao necessariamente 0\% -- sempre impresso,
      nunca assumido zero.
  \end{itemize}
  \bk
\end{frame}

\begin{frame}{Quando seria o MIP a escolha certa?}
  \small Mesmo nesta classe de problemas, o MIP \'e a escolha mais
  natural em situa\c{c}\~oes espec\'ificas:
  \begin{itemize}
    \item \textbf{Apenas atribui\c{c}\~ao de dia, sem horas exactas.}
      Quando o resultado \'e ``caso $c$ no dia $d$ no bloco $r$'' sem
      hora de in\'icio, o MIP de balde de dia captura isto naturalmente.
    \item \textbf{Sem recursos partilhados entre blocos.} Se cada
      equipamento \'e dedicado a um bloco e os cirurgi\~oes n\~ao mudam
      de bloco, a contagem di\'aria \'e suficiente.
    \item \textbf{Sem restri\c{c}\~ao multi-dia a jusante (C11).} Sem
      modelar camas de recupera\c{c}\~ao, desaparece uma raz\~ao
      estrutural chave para intervalos.
    \item \textbf{Inst\^ancia pequena, relaxa\c{c}\~ao LP ajustada.}
      Um solver comercial (Gurobi, CPLEX) pode fechar o desfasamento
      MIP em segundos -- mais r\'apido que CP-SAT em inst\^ancias
      pequenas.
    \item \textbf{Necessidade de vari\'aveis duais LP.} Pre\c{c}os-sombra
      suportam an\'alise de sensibilidade -- o CP-SAT n\~ao os expe.
    \item \textbf{Equipa com experi\^encia em MIP.} O risco de codificar
      incorrectamente restri\c{c}\~oes \'e menor com a ferramenta
      que a equipa conhece melhor.
  \end{itemize}
  \bk
\end{frame}

\begin{frame}{MIP -- vis\~ao geral da formula\c{c}\~ao}
  \hypertarget{mip1}{}
  Implementado em \texttt{milp\_baseline\_solver.py}; executa via
  \texttt{--solver milp-cbc} (inclu\'ido), \texttt{milp-gurobi}, ou
  \texttt{milp-cplex} (licen\c{c}a comercial necess\'aria).

  \textbf{Igual ao CP-SAT:} conjuntos, pr\'e-filtro de elegibilidade
  (C4--C6), prioridades, $w_c$ do mesmo \texttt{penalty.py}.

  \textbf{Diferen\c{c}as principais:}
  \begin{itemize}\small
    \item Sem horas exactas intra-dia -- decis\~oes ao n\'ivel
      dia+bloco apenas.
    \item C7 e C10 s\~ao somas de capacidade lineares, n\~ao
      verifica\c{c}\~oes exactas de sobreposi\c{c}\~ao temporal.
    \item C11 (camas de recupera\c{c}\~ao) n\~ao \'e express\'avel --
      uma vari\'avel de balde de dia n\~ao cont\'em o ``dia de
      cirurgia'' como valor.
    \item Sem Termo 4 no objectivo (C11 n\~ao existe nesta formula\c{c}\~ao).
  \end{itemize}
  \bk
\end{frame}

\begin{frame}{MIP -- vari\'aveis}
  \footnotesize
  \begin{center}
  \begin{tabular}{p{2.2cm}p{2.5cm}p{6.9cm}}
    \toprule
    \textbf{Vari\'avel} & \textbf{Dom\'inio} & \textbf{Significado} \\
    \midrule
    $x_{cdr}$ & $\{0,1\}$ &
      1 se o caso $c$ \'e agendado no dia $d$ no bloco $r$.
      Apenas criada para trios eleg\'iveis.
      Determina o dia e o bloco, mas \emph{n\~ao} a hora exacta. \\[3pt]
    $z_c$ & $[0,1]$ &
      1 se o caso $c$ n\~ao \'e agendado esta semana (s\'o para
      $p_c\ne4$). Declarada cont\'inua -- C3 for\c{c}a-a a $\{0,1\}$
      no \'optimo porque o objectivo penaliza n\~ao-agendamento
      fraccion\'ario. \\
    \bottomrule
  \end{tabular}
  \end{center}
  \vspace{0.3em}
  \textbf{Objectivo} (mesma estrutura de tr\^es termos que o CP-SAT,
  sem Termo 4):
  \[\footnotesize
    \min\;\sum_{c:\,dd_c\ge0}\sum_{d,r}[dd_c+d]\,x_{cdr}
    +\sum_{c:\,dd_c<0}\sum_{d,r}[dd_c+\alpha d]\,x_{cdr}
    +\sum_{c:\,p_c\ne4}w_c\,z_c
  \]
  \bk
\end{frame}

\begin{frame}{MIP -- restri\c{c}\~oes e o que est\'a em falta}
  \textbf{C7 (bloco, soma agregada):}
  \[\small
    \sum_{c\in C}t_c^{tot}\,x_{cdr}\le k_{dr}\qquad\forall d,r
  \]
  Para um \'unico bloco: equivalente \`a n\~ao-sobreposi\c{c}\~ao. Para
  um \emph{recurso partilhado entre blocos}: trata incorrectamente
  todos os usos do equipamento como simult\^aneos.

  \textbf{C10 (equipamento, contagem di\'aria):}
  \[\small
    \sum_{c:\,u_{ce}=1}\sum_r x_{cdr}\le\kappa_{ed}\qquad\forall e,d
  \]
  Conta \emph{quantos} casos do equipamento $e$ caem num dia, n\~ao se
  se sobrep\~oem -- principal fonte do desfasamento de 5.56\%.

  \textbf{Sem C11:} ``$\text{bed}_c$ inicia no dia em que $c$ \'e
  agendado'' n\~ao pode ser escrito -- nenhuma vari\'avel num modelo de
  balde de dia guarda o ``dia de cirurgia'' como \emph{valor}.
  \bk
\end{frame}

  \small Mantido por uma raz\~ao: \textbf{limpeza dependente da
  sequ\^encia} -- uma matriz de transi\c{c}\~ao entre casos adjacentes
  num bloco via \texttt{sequence\_var + no\_overlap(transition\_matrix)}.
  \begin{center}\small
  \begin{tabular}{lcc}
    \toprule
    & Mesmo servi\c{c}o & Servi\c{c}o diferente \\

    CP-SAT (por escal\~ao de dura\c{c}\~ao) & $t_c^{clean}$ do pr\'oprio caso & igual \\

  \end{center}
  Inst\^ancia demo: id\^entico ao CP-SAT (155.0, 20/20, \'Optimo).

  Com 200 casos, o pr\'oprio desfasamento fecha mais lentamente do que

  n\~ao uma fraqueza de modelagem).
  \bk
\end{frame}
